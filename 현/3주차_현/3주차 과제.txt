# 1번 홈 화면 API

# 1-1 지역 드롭다운 선택 API
Endpoint: PUT /api/me/location # me는 alias처럼 쓰여서 userId값을 반환
Request Header: Authorization : Bearer {access_token}
Path Variable: userId
Query String: 없음
Request Body: 
{”location”: “안암동”}

# 1-2 미션 달성도 API
Endpoint: GET /api/home/mission-summary
Request Header: Authorization : Bearer {access_token}
Path Variable: 없음
Query String: location=안암동
Request Body: 없음 # 받아올 실제 데이터가 없어서

# 1-3 내 미션 보기 API
Endpoint: GET /api/home/my-missions
Request Header: Authorization : Bearer {access_token}
Path Variable: 없음
Query String: status=ONGOING # 실제로는 0
page=1&size=10
Request Body: 없음

# 1-4 미션 도전 버튼 API
Endpoint: POST /api/missions/{missionId}/start # 시작 버튼 누르면 시작
Request Header: Authorization : Bearer {access_token}
Path Variable: missionId
Query String: 없음
Request Body: 없음

# 3번 미션 목록 조회 API
